#include <iostream>

int main() {
    int array[5]; // Fixed-size array
    std::cout << "Enter 5 numbers:\n";
    for (int i = 0; i < 5; ++i) {
        std::cin >> array[i]; // Unsafe input operation
    }

    // Print the numbers
    std::cout << "You entered:\n";
    for (int i = 0; i < 5; ++i) {
        std::cout << array[i] << "\n"; // Unsafe output operation
    }

    return 0;
}
